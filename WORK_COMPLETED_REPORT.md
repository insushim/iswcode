### Challenge 구조 샘플
```typescript
{
  id: 'w31-m2-c1',
  title: '연습 1: h1 태그로 제목 만들기',
  description: '"나의 웹사이트"라는 제목을 h1 태그로 만드세요',
  difficulty: 'easy',
  starterCode: '<!-- h1 태그로 제목을 만드세요 -->\n',
  solution: '<h1>나의 웹사이트</h1>',
  testCases: [
    {
      input: '',
      expectedOutput: '<h1>나의 웹사이트</h1>',
      description: 'h1 태그 확인',
      errorHint: '<h1>과 </h1>로 감쌌나요?'
    },
    {
      input: '',
      expectedOutput: /^<h1>.+<\/h1>$/,
      description: 'h1 구조',
      isHidden: true
    }
  ],
  hints: [
    '<h1>내용</h1> 형식',
    '닫는 태그에는 /가 필요해요'
  ],
  estimatedMinutes: 2,
  feedback: {
    perfect: '완벽해요! h1 태그를 정확히 사용했어요!',
    good: '거의 다 됐어요! 태그를 확인해보세요.',
    partial: '태그 구조는 맞는데 내용을 확인해보세요.',
    wrong: '<h1>과 </h1>로 내용을 감싸야 해요.',
    syntaxError: '태그 문법을 확인하세요. <>가 맞나요?'
  },
  commonErrors: [
    {
      pattern: '<h1>[^<]*$',
      message: '닫는 태그가 없어요!',
      suggestion: '</h1>을 추가하세요'
    },
    {
      pattern: '<H1>',
      message: 'HTML은 소문자가 권장돼요',
      suggestion: '<H1> → <h1>'
    }
  ]
}
```
